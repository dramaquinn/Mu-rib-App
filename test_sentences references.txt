يطير الطائر في السماء 
يعزف أحمد بالناي
يأكل محمود البيتزا
سلمَ أحمدُ الكرةَ لعبدِ اللهِ
لا يجبُ أنْ تأكلَ الطعامَ بشمالِكَ
هبطتِ الطائرةُ في المطارِ
حطمتِ الفتاةُ الصحنَ عندما وقعَ
أُرسلَ الرسولُ محمدٌ بالحقِ من ربِهِ
حفظَ الصحابةُ القرآنَ
حلَّ المهندسُ المشكلةَ
رمى الولدُ القمامةَ
صممَ المهندسُ العمارةَ
أكلتُ الطعامَ الفاخرَ في إسبانيا
رقعةُ الفتوحاتِ الإسلاميةِ كانتْ كبيرةً وواسعةً
امتازَ عهدُ العثمانيينَ بالتطورِ العلميِّ
ينصحُ الناجحونَ بمطالعةِ الكتبِ
إنجازاتُ الطالبِ الجامعيِّ ليستْ كثيرةً
صعدَ الفارسُ على ظهرِ الخيلِ
حدثني أخي عن مباراةِ البارحةِ
دفعتُ المالَ إلى التاجرِ
اختفتِ الشمسُ
ذهبَ خلفَ المبنى
بقيتُ في المنزلِ
جلسَ صديقي بالقربِ من النافذةِ
جدفتُ القاربَ حتى أصلَ إلى الجزيرةِ
ركبتُ القطارَ إلى الجامعةِ
رأيتُهُ يمشي وحيداً البارحةَ
صببتُ الماءَ في قنينةٍ
شربتُ الماءَ من كأسٍ
صافحتُ أخي
تركتُ الجامعةَ حتى أكسبَ المالَ
أكلَ مصعبٌ التفاحةَ
التقى الرجلُ أخاهُ في المطارِ
تحدثتُ معَ المسافرِ بالإنجليزيةِ
أودُّ أنْ أتخرجَ من الجامعةِ
عاملِ الحيواناتِ برفقٍ
اعتنى الولدُ بوالدِهِ
الأطفالُ كثيرو العبثِ
قرأتُ المصحفَ
أخفى الجاسوسُ نواياهُ
آثرَ أخوهُ على نفسِهِ
صادقَ الطالبُ زميلَهُ
أُصيبَ الرجلُ في قلبِهِ
فشلَ الطالبُ في الامتحانِ
كانَ أولُ البشرِ آدمَ عليهِ السلامُ
كافأَ المعلمُ طالبَهُ
تُركتِ المرأةُ في حجرتِها
الشعبُ الفلسطينيُّ في صراعٍ مستمرٍ
أحبُّ البقولياتِ
الامتحانُ كانَ يسيراً
استُشهدَ جدي في معركةٍ
ذهبَ الطالبُ إلى الجامعةِ
أكلتُ البرتقالةَ
تجمدَ الماءُ في الثلاجةِ
سددَ المهاجمُ الكرةَ
دفعتُ للرجلِ الإيجارَ
أرسلتُ لصديقي رسالةً
اعتزلتُ لعبَ الكرةِ بعدَ الإصابةِ
استخدمتُ المصباحَ لإنارةِ الظلامِ
تأخرتُ عن عملي بسببِ الازدحامِ
فوجُ المتخرجينَ هذا العامَ كبيرٌ
أسعدَني فوزُ المغربِ
شوقي إلى الأقصى كبيرٌ
انتشرتِ الأوبئةُ في الفترةِ الأخيرةِ
تحومُ الأرضُ حولَ الشمسِ
الولدُ مفخرةُ أبيهِ
يأكلُ الرجلُ الخيارَ
ينحتُ الفنانُ التمثالَ
قطعَ الرجلُ شجرةً باستخدامِ المنشارِ
أكلَ التمساحُ السمكَ
يعزفُ الفنانُ الجيتارَ
يعجنُ الخبازُ عجينةَ الخبزِ
نامَ الرجلُ على سريرِهِ
أنكرَ الخاسرُ خسارتَهُ
ترددتُ في الإجابةِ على الهاتفِ
تُباعُ الحاجاتُ العتيقةُ بمبالغَ كبيرةٍ
الأحجارُ النفيسةُ باهظةُ الثمنِ
تعطلتِ الآلةُ
الوقايةُ خيرٌ من العلاجِ
ماءُ النهرِ صالحٌ للشربِ
ضبطتُ ساعةَ المؤقتِ
أبرزَ الرجلُ هويتَهُ على الحاجزِ
أبرمَ الرئيسُ اتفاقَ هدنةٍ مع العدوِ
النظافةُ من الإيمانِ
طردَ الحكمُ اللاعبَ
أعانَ اللهُ الأبكمَ
أبطلَ الرئيسُ الهدنةَ
أبلغتُ والدي بنجاحي
تقدمتْ دولتُنا في التكنولوجيا بفضلِ شبابِها
أعلنَ الوزيرُ أسماءَ العشرةِ الأوائلِ
امتلأتْ أركانُ الغرفةِ بالناسِ
سُمِعَ خبرُهُ في جميعِ أنحاءِ الدولةِ
رفرفتِ الطيورُ في السماءِ
سآخذُ قسطاً من الراحةِ
كادَ أنْ يقعَ على الأرضِ
نيتُهُ صافيةٌ
قرأتُ تغريدةً على مواقعِ التواصلِ الاجتماعيِّ
يحبُّ الأطفالُ اللعبَ واللهوَ
كانَ الأسيرُ هزيلَ الجسدِ
اقتنيتُ أحدثَ المعداتِ الرياضيةِ
صرخَ الصبيُّ الماءَ
أهملَ الأبُ ابنَهُ
لا يستطيعُ الإنسانُ العيشَ إلى الأبدِ
درستُ للامتحانِ دونَ جدوى
الفولاذُ معدنٌ صلبٌ
عملتُ بجدٍ على مشروعي
عاملَ الأبُ ابنَهُ بلطفٍ
أنا أحلمُ بتحقيقِ كلِّ غاياتي
شُيِّدتْ في هذا العصرِ مبانٍ كبيرةٌ
هلكَ فرعونُ
غُلبتِ الرومُ
انتشرَ الثأرُ في الباديةِ
توارى المناضلُ عن الأنظارِ
الرئيسُ شخصٌ ثريٌّ
الملكُ هو صاحبُ أعلى سلطةٍ في الدولةِ
ماءُ العينِ مالحٌ
نفدتِ البضاعةُ من عندِنا
علمَ الأستاذُ الدرسَ للطلابِ
عُقدَ الاجتماعُ في العاصمةِ
ضربَ الأخُ أخاهُ
إنَّ الابتسامةَ لا تكلفُ شيئاً ولكنَّها تعني الكثيرَ
كانوا قديماً يدرسونَ في المسجدِ
شعرَ رائدٌ بالتحطمِ بعدَ خسارتِهِ
الشعرُ من جمالياتِ الإنسانِ
انحسرتْ مستوياتُ البحرِ
خسرَ الصبيُّ كأسَ البطولةِ
جاءَنا البيانُ الآتي
الإضافاتُ الجديدةُ على التطبيقِ مفيدةٌ
الخلافاتُ ستسببُ المشاكلَ
انتقدَ الصحفيُّ الحكومةَ
الأسدُ من أشرسِ حيواناتِ الغابةِ
مياهُ النهرِ عذبةٌ بطبيعتِها
توسلَ بسامٌ إلى ربِهِ بالرحمةِ
الحديدُ ليسَ بالمعدنِ المرنِ
لكلِّ جملةٍ معنىً خفيٌّ
غزا الجيشُ الأعداءَ
فاضتِ الجداولُ عن آخرِها بالمياهِ
يمتلكُ أخي قناعاتٍ متحجرةً
العملاتُ الحديديةُ كانتْ مشهورةً قديماً
كانَ عددُ الطلابِ في الفصلِ عشرينَ طالباً
التقطتُ الثمرةَ من على الشجرةِ
امتطيتُ حصاني وانطلقتُ في الطريقِ مسرعاً
اللهمَّ أبعدْ عنا السنواتِ الرملاءَ
ذهبَ عني النومُ الليلةَ الماضية
نحنُ محظوظونَ أنَّنا نعيشُ في زمنٍ يكثرُ فيهِ الأدويةُ
يوجدُ في الإسلامِ أربعةُ مذاهبَ فقهيةٍ
الوقايةُ أفضلُ علاجٍ للمرضِ
 يسعدُني أنْ تحفظَ القرآنَ
 يبدو أنَّ المباراةَ ستكونُ قويةً
 لن يهملَ الطبيبُ مرضاهُ
 يعملُ المهندسُ كي يطورَ المشروعَ
 تجلسُ البنتُ لتذاكرَ دروسَها
 المدرستان الحكوميتانِ متطورتانِ
 البحثُ الأكاديميُّ دقيقٌ
 المدينةُ التاريخيةُ جميلةٌ
 الموظفونَ المتميزونَ مكرمونَ
 القصةُ القصيرةُ معبرةٌ
 صفحاتُ كتابِ التاريخِ مشوقةٌ
 بينَ سطورِ الكتابِ معانٍ جميلةٌ
 محبو قراءةِ الشعرِ أدباءٌ
 فوقَ قممِ الجبالِ هواءٌ نقيٌّ
 أصحابُ مصانعِ النسيجِ تجارٌ
 سمعتُ الخبرَ  
 كتبتُ الرسالةَ  
 ركبَ السيارةَ  
 غسلَ الكوبَ  
 أعدتْ الدرسَ  
 جمعتْ الأوراقَ  
 نظفنَ البيتَ  
 رتبنَ الغرفةَ
 فتحا البابَ  
 حملا الصندوقَ  
 سافرَ الأميرُ خالدٌ  
 حضرَ المديرُ سالمٌ  
سكنتُ في البيتِ بل الشقةِ  
 أكلتُ التفاحَ بل البرتقالَ  
المهندستانِ هندُ وليلى ماهرتانِ  
قرأتُ الكتابَ بل المجلةَ  
تعطلتْ السيارةُ محركُها
زرتُ المدرسةَ بل الجامعةَ  
 سُرِقَ الكتابُ 
غُسِلَ الثوبُ 
حُفِظَ السرُ 
شُرِبَتْ القهوةُ 
قُطِفَتْ الزهرةُ 
حصدَ الجيشُ النصرَ
قدَّمَت الممرضتانِ الأدويةَ
صممَ المهندسونَ المشروعَ
قادَ القائدُ الجيشَ
كانت البائعاتُ لطيفاتٍ  
كانت السياراتُ جديدةً  
أصبح الطريقُ معبداً  
أصبحت القصةُ ممتعةً 
أضحى ذو المالِ كريماً  
أضحى من المهمِ أن يتعلمَ الشبابُ التكنولوجيا
أمسى البحرُ هادئاً  
أمسى أخو أسعدٍ طبيباً  
أمسى من البديهيِ أنَّ التطورَ التكنولوجيَ لا يتوقفُ  
ظلَّ القمرُ مضيئاً طوالَ الليلِ  
ظلَّ الماءُ متدفقاً في النهرِ  
ليس الخبزُ طازجاً  
ليس الوقتُ متأخراً  
ما دام النورُ ساطعاً، فالطريقُ واضحٌ
ما دام الوقتُ مبكراً، فالفرصةُ سانحةٌ
ما زالت الحديقةُ خضراءَ في الخريفِ
ما زال الباحثونَ يدرسونَ الظاهرةَ
ما انفكت الأمةُ تتطلعُ إلى مستقبلٍ أفضلَ
ما انفك القائدُ يوجهُ جنودَه بحكمةٍ
ما برحت الحضارةُ تتقدمُ عبرَ العصورِ
ما برح النورُ منبعثاً من المصباحِ
ما فتئت الحقيقةُ تظهرُ مع الوقتِ
إنَّ الصدقَ منجاةٌ
كأنَّ الوردَ أحلامٌ ملونةٌ
أدركتُ أنَّ السعادةَ في القناعةِ
ليتَ السلامَ يعمُّ العالمَ
لعلَّ الفرحَ يملأُ القلوبَ
تعزفُ الموسيقيتانِ اللحنَ
يطبخُ الطباخانِ الحساءَ
تنسقُ المضيفاتُ الرحلاتِ
نحصدُ نحنُ الثمارَ
تنحتونَ أنتم التماثيلَ
تربيانِ أنتما الدواجنَ
هربَ اللصُ مِن السجنِ إلى الجبلِ
نقلَ النحالُ العسلَ مِن الخليةِ
كشفَ الغواصُ عن اللؤلؤِ
لم يثمرْ الشجرُ بعدُ في هذه الحديقةِ
لا تُفرطْ في استخدامِ الهواتفِ الذكيةِ ليلاً
عزفَ الموسيقيُ عن الحفلةِ
رحبَ المضيفُ بالضيفِ
تمسكَ المتزلجُ بالحبلِ
استلقى السباحُ على الشاطئِ
علقَ الرسامُ اللوحةَ على الجدارِ
لم يظهرْ أيُّ دليلٍ على صحةِ الفرضيةِ
لا تسرفْ في الحديثِ عن أمورٍ شخصيةٍ مع الغرباءِ
لِيُتقنْ الطهاةُ أطباقَهم لتنالَ إعجابَ الزبائنِ
لِيَبحثْ العالمُ عن حلولٍ جديدةٍ للتحدياتِ البيئيةِ
إنْ تُحسنْ تدبيرَ أموالك تضمنْ مستقبلًا ماليًّا مستقراً
إنْ تقرأْ في علومِ النفسِ تفهمْ نفسكَ أكثرَ
مَنْ يجمعْ الكتبَ ينمِّ ثقافتَه باستمرارٍ
مَنْ يسهرْ الليلَ في العملِ يشعرْ بالتعبِ في النهارِ
متى تُنظمْ أفكارَك تكونْ قراراتُك أفضلَ
متى تبذلْ قصارى جهدِك تصلْ إلى مبتغاكَ
أينما تكتشفْ مجالاتٍ جديدةً تتعلمْ مهاراتٍ متنوعةً
أينما تتواجدْ في الأماكنِ العامةِ تلتقِ بأناسٍ جددٍ
كيفما تُعبرْ عن أفكارِك تصلْ إلى قلوبِ الآخرينِ
كيفما تُعدْ طعامَك تستمتعْ به أكثرَ
ما تبذلْ من جهودٍ الآن تجنِ ثمارَها مستقبلاً
ما تكتبْه من ملاحظاتٍ يساعدْك في تذكرِ التفاصيلِ
يَّ صديقٍ تُحسنْ معاملتَه يصبحْ عوناً لك
مهما تتعلمْ من لغاتٍ تبقَ اللغةُ الأمُّ الأقربَ لقلبِك
مهما تبدو لك الحياةُ معقدةً، اجعلْ الصبرَ رفيقَك
في أيِّ مجالٍ تخترْ أن تتخصصْ تصلْ إلى درجةِ الإتقانِ
في أيِّ طريقٍ تتخذْ طريقاً للوصولِ تكتسبْ تجربةً جديدةً
تجولَ السائحُ ظهراً بينَ المعالمِ السياحيةِ
ذهبَ الأطفالُ شرقاً صباحاً للاستكشافِ
ارتاحَ العاملُ تحتَ ظلِ الشجرةِ ظهيرةً
غادرتُ المكتبَ مساءً بعدَ انتهاءِ العملِ
قامَ المزارعُ بالعملِ حولَ الأرضِ طوالَ النهارِ
